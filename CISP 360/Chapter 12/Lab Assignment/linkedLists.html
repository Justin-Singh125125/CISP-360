#include <iostream>

using namespace std;

//creates a single node when pointed too
struct ListNode
{
	int data;
 	struct ListNode *next;
};
ListNode *addToListEnd(ListNode *list, int d)
{
	ListNode *newNode = new ListNode;
	ListNode *temp = new ListNode;
	//put whatever d is in the new node
	newNode->data = d;
	newNode->next = NULL;
	if (list == NULL)
	{
		temp = newNode;

		
	}
	else
	{
		//moves the pointer temp right before the end of the linked list
		temp = list; 
		while (temp->next != NULL)
		{
			
			temp= temp->next;
			
		}
		temp->next= newNode;
		
	}

	
	
	
	return temp;

	
}
	
	
	
void displayLinkedList(struct ListNode *head)
{
	struct ListNode *p = head;
	while (p != NULL)
	{
		cout << p->data << " ";
		p = p->next;
	}
	
}


int main()
{
	
	//create a pointer that is going to be the head
	ListNode *myList = NULL;
	//a copy pointer
	ListNode *myListCopy = NULL;
	
	//call a function to add nodes to the end of the list
	myList = addToListEnd (NULL,1);
	//append 3 to the list

	myList = addToListEnd(myList,4);
	displayLinkedList (myList);

}
